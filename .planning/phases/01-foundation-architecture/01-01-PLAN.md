Create Figma plugin manifest and typed message protocol:

   - id: Generate a unique plugin ID (use format: "screenshot-to-shadcn-{random-8-chars}")
   - api: "1.0.0"
   - main: "dist/main.js"
   - ui: "dist/ui.html"
   - editorType: ["figma"] (not figjam)
   - documentAccess: "dynamic-page"
   - networkAccess: { allowedDomains: ["https://api.anthropic.com"] }

2. src/shared/messages.ts:
   Define typed message protocol with correlation IDs for async request/response:

   ```typescript
   // Message from main thread to UI
   export type PluginMessage =
     | { type: 'INIT'; correlationId: string }
     | { type: 'SELECTION_CHANGED'; correlationId: string; nodeCount: number }
     | { type: 'RESPONSE'; correlationId: string; payload: unknown };

   // Message from UI to main thread
   export type UIMessage =
     | { type: 'UI_READY'; correlationId: string }
     | { type: 'REQUEST'; correlationId: string; action: string; payload?: unknown }
     | { type: 'CLOSE_PLUGIN' };

   // Helper to generate correlation IDs
   export function generateCorrelationId(): string {
     return `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
   }

   // Type guard helpers
   export function isPluginMessage(msg: unknown): msg is PluginMessage {
     return typeof msg === 'object' && msg !== null && 'type' in msg;
   }

   export function isUIMessage(msg: unknown): msg is UIMessage {
     return typeof msg === 'object' && msg !== null && 'type' in msg;
   }
   ```

   This pattern enables:
   - Main sends REQUEST with correlationId
   - UI responds with same correlationId
   - Caller can await specific response
  </action>
